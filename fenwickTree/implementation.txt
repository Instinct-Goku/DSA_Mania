#include<bits/stdc++.h>
using namespace std;

vector<int> fenwick;

void constructUtil(int ind, int value, int n) {
    ind = ind+1;

    while(ind <= n) {
        fenwick[ind] += value;
        ind += (ind&(-ind));
    }
}

void constructFenwick(vector<int> &v) {
    int n = v.size();
    fenwick.resize(n+1,0);

    for(int i = 0; i < n; i++) {
        constructUtil(i,v[i],n);
    }
}

int getSum(int ind) {
    int sum = 0;

    while(ind) {
        sum += fenwick[ind];
        ind -= (ind&(-ind));
    }

    return sum;
}

int getSumInRange(int l, int r) {
    return getSum(r) - getSum(l-1);
}

int main() {
    vector<int> v = {1,3,2,5,9,4};
    constructFenwick(v);
    
    cout<<getSum(5)<<" ";

    cout<<getSumInRange(4,5);
}
