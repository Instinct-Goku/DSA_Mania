#include<bits/stdc++.h>
using namespace std;

void add(int *l1, int *l2, int m, int n){
    int res[20];
    int carry = 0;
    int i = m;
    int j = n;
    int k = m;
    int t = 0;
    while(j >= 0){
        t = l1[i] + l2[j] + carry;
        res[k] = t%10;
        carry = t/10;
        i--;
        j--;
        k--;
    }
    
    while(i >= 0){
        t = l1[i] + carry;
        res[k] = t%10;
        carry = t/10;
        k--;
        i--;
    }
    
    if(carry)
        cout<<carry<<" ";
        
    for(int i = 0; i <= m; i++)
        cout<<res[i]<<" ";
  
    
}

int main() {
    int l1[] = {1,5,4,8};
    int l2[] = {9,2,1,5};
    int m = sizeof(l1)/sizeof(l1[0]) - 1;
    int n = sizeof(l2)/sizeof(l2[0]) - 1;
    
    if(m > n)
        add(l1, l2, m, n);
    else if(n > m)
        add(l2, l1, n, m);
    else
        add(l1, l2, m, n);
}
